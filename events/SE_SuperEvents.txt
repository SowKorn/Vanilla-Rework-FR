###1939###
# World War 2
country_event = {
	id = superevents.103
	title = superevents.0.t
	desc = superevents.0.d
	
	immediate = {
		set_global_flag = SE_superevents_activate
		set_global_flag = SE_world_war_ii
		
		set_global_flag = SE_prevent_duplicate_wwii
		#cue funni music
	}

	immediate = {
		sound_effect = ww2_begin
	}

	hidden = yes
	
	fire_only_once = yes

	trigger = {
		NOT = {
			has_global_flag = SE_superevents_activate
		}
		GER = {
			is_major = yes
			is_subject = no
		}
		any_country = {
			is_major = yes
			has_war_with = GER
			NOT = {
				original_tag = GER
			}
		}
		NOT = {
			has_global_flag = SE_prevent_duplicate_wwii
		}
	}
	
	option = {
		name = superevents.0.a
	}

country_event = {
	id = superevents.104
	title = superevents.0.t
	desc = superevents.0.d
	
	immediate = {
		set_global_flag = SE_superevents_activate
		set_global_flag = SE_gerpol_war

		set_global_flag = SE_prevent_duplicate_wwii
		#cue funni music
	}

	immediate = {
		sound_effect = ww2_begin
	}

	hidden = yes
	
	fire_only_once = yes

	trigger = {
		NOT = {
			has_global_flag = SE_superevents_activate
		}
		any_country = {
			is_major = yes
			NOT = {
				OR = {
					has_war_with = GER
					has_guaranteed = POL
					is_in_faction_with = POL
				}
			}
		}
		GER = {
			is_subject = no
			has_offensive_war_with = POL
		}
		POL = {
			is_subject = no
		}
		NOT = {
			has_global_flag = SE_prevent_duplicate_wwii
		}
	}
	
	option = {
		name = superevents.0.a
	}
}
